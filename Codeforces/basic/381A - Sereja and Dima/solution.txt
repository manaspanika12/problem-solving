#include <bits/stdc++.h>
using namespace std;

int main() {
	
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++){
	    cin>>arr[i];
	}
	int p1=0;
	int p2=n-1;
	
	int s=0;
	int d=0;
	
	int i=0;
	while(p1<=p2){
	    if(i%2==0){
	        if(arr[p1]>arr[p2]){
	            s+=arr[p1];
	            p1++;
	        }
	        else{
	            s+=arr[p2];
	            p2--;
	        }
	    }
	    else{
	        if(arr[p1]>arr[p2]){
	            d+=arr[p1];
	            p1++;
	        }
	        else{
	            d+=arr[p2];
	            p2--;
	        }
	    }
	    i++;
	}
	cout<<s<<" "<<d;
    return 0;
    
}
