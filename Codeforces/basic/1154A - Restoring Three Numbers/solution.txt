#include <bits/stdc++.h>
using namespace std;

int main() {
	
	int num[4];
    
    for(int i=0;i<4;i++){
        cin>>num[i];
    }
    int max=0;
    
    for(int i=1;i<4;i++){
        if(num[max]<num[i]){
            max=i;
        }
    }
    
    for(int i=0;i<4;i++){
        if(i!=max){
            cout<<(num[max]-num[i])<<" ";
        }
    }
	
    return 0;
    
}
