#include <bits/stdc++.h>
using namespace std;
int charToInt(char c) {
    return c - '0';
}
int main() {
	
	int t;
	cin>>t;
	
	while(t--){
	    string s;
	    cin>>s;
	    int ls = 0, rs = 0;
	    
	    for(int i=0;i<3;i++){
	        ls+=charToInt(s[i]);
	        rs+=charToInt(s[5-i]);
	    }
	    if(ls==rs){
	        cout<<"Yes\n";
	    }
	    else{
	        cout<<"No\n";
	    }
	}
	
    return 0;
    
}
